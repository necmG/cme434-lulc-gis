pixels = pd.read_csv('/content/drive/MyDrive/all_pixels_prediction.csv')

coords = pixels['.geo'].apply(lambda x: json.loads(x)['coordinates'])
lons = coords.apply(lambda x: x[0])
lats = coords.apply(lambda x: x[1])

X_pred = pixels.drop(
    columns=['system:index', '.geo'] + drop_cols,
    errors='ignore'
)
X_pred = X_pred.fillna(X_pred.median())

# Sütun sırasını eğitimle eşleştir
missing = set(X_train.columns) - set(X_pred.columns)
extra   = set(X_pred.columns) - set(X_train.columns)
if missing: print('⚠️  Eksik:', missing)
if extra:   print('⚠️  Fazla:', extra)

X_pred = X_pred[X_train.columns]

probabilities = xgb_model.predict_proba(X_pred)[:, 1]
predictions   = xgb_model.predict(X_pred)

result = pd.DataFrame({
    'longitude':   lons,
    'latitude':    lats,
    'probability': probabilities,
    'prediction':  predictions
})

print(f'Toplam piksel:  {len(result)}')
print(f'Değişmiş alan:  {predictions.mean():.1%}')

result.to_csv('/content/drive/MyDrive/lulc_change_map.csv', index=False)
print('Kaydedildi.')
